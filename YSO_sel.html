def sel_criteria(filename,columns,Nite=4,intl=[0,16],plot=False): #Vphas_criterio_seleccion.cat
    """
    Columns = [r-i,r-ha,r,Ra,Dec]
    """
    from scipy import stats
    from math import isnan
    #isnan retorna verdadero si el argumento es nan.

    c = np.genfromtxt(filename,usecols=columns) #11->r_i, 12->r_ha, 68->r, 2->Ra, 3->Dec.
    #Quitamos los nan de la columna de la magnitud r:
    c = np.array([(c[i]) for i in range(len(c[:,0])) if isnan(c[i,2])==False])
    #Seleccionamos el intervalo de r:
        
    c = np.array([(c[i]) for i in range(len(c[:,0])) if intl[0]<c[i,2]<intl[1]])
    #Hacemos la selección para el factor multiplicador de clip:
    r = c[:,2]
    ri = r[np.argsort(r)]
    ri = ri[-1]
    
    #print ri,rf
    #Quitamos los nan de las diferencias de colores individualmente, porque no pude quitarlas las dos de una :(
    c = np.array([(c[i]) for i in range(len(c[:,0])) if isnan(c[i,0])==False]) #r-i
    c = np.array([(c[i]) for i in range(len(c[:,1])) if isnan(c[i,1])==False]) #r-Ha
    
    r_i_init = c[:,0]
    r_ha_init = c[:,1]
    
    r_i = r_i_init[:]
    r_ha = r_ha_init[:]
        
    for i in range(Nite):
        # Normal linear fit
        (slope, inter, corr, p, sigma) = stats.linregress(r_i, r_ha)

        # Standard error of the fit:
        line_fit = slope*r_i + inter
        delta = r_ha - line_fit
        std_err = np.sqrt(sum((delta)**2)/len(r_i))
        
        if i == 0:
            line_fit_init = line_fit
        if i == Nite-1:
            if ri <= 17:
                factor = 2*clip
                break
            elif ri > 17 and ri <= 19:
                factor = 3*clip
                break
            else:
                factor = 4.5*clip
                break   
        
        #Si se quieren ver todos los fits de las iteraciones, descomentar las siguientes líneas:
        #plt.plot(r_i,r_ha,'r.')
        #plt.plot(r_i,line_fit,'g-')

        clip = std_err #just 1 sigma
        
        #Sólo sacamos los puntos que están bajo el ajuste:
        c_new = np.array([c[i] for i in range(len(r_i)) if -delta[i] < clip])
        
        r_i_new = c_new[:,0]
        r_ha_new = c_new[:,1]

        #plt.plot(r_i_new,r_ha_new,'p')
        c = c_new
        r_i = r_i_new
        r_ha = r_ha_new
        
    if plot == True:
        plt.figure(figsize=((12,7)))
        plt.xlabel('r-i')
        plt.ylabel('r-ha')
        plt.plot(r_i_init,r_ha_init,'r.',label="All data") #puntos iniciales
        plt.plot(r_i_init,line_fit_init,'r-',label="Initial fit") #fit inicial
        plt.plot(r_i,r_ha,'o', mfc='none',label='Final data') #puntos usados para hacer el fit final
        plt.plot(r_i,line_fit,'g-',label="Final fit") #fit final
        plt.plot(r_i,line_fit+factor,'k-',label="Final cut") # fit final más el factor clip
        plt.legend(loc='best', numpoints=1)
        
    #Seleccionamos las estrellas que están sobre la linea:
    Ra = c_new[:,3]
    Dec = c[:,4]
    r = c[:,2]
    sources = np.array([(Ra[i],Dec[i],r[i],r_i[i],r_ha[i]) for i in range(len(r_i)) if r_ha[i]>slope*r_i[i]+inter+factor]) 
    return sources
